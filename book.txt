function getTimeIn12HourFormat() {
  const now = new Date();
  const hours = now.getHours();
  const minutes = now.getMinutes();
  const ampm = hours >= 12 ? 'PM' : 'AM';
  const twelveHourFormatHours = hours % 12 || 12; // Convert 0 to 12

  const formattedTime = `${twelveHourFormatHours}:${minutes.toString().padStart(2, '0')} ${ampm}`;
  return formattedTime;
}

const formattedTime = getTimeIn12HourFormat();
console.log(formattedTime);
